3 Promise，在request下创建index.js

/* es6对外部暴露一个函数为request函数 */
export const request = (params) => {
  return new Promise((resolve, reject) => {
    wx.request({
      ...params,
      success: (result) => {
        resolve(result)
      },
      fail: err => reject(err),
    });
  })
}

      padding: 10rpx 0;

        }

        border-left: 10rpx solid #fff;

        image {
          width: 100%;
          height: 100%;
        }
      }
    }
  }
}
