# 箭头函数

  // 1. es6允许使用箭头 （ => ） 定义函数
  // let fn = function () {

  // }

  // let fn = (a, b) => {
  //   return a+b
  // }
  // console.log( fn(1,2) )

  // 1. this是静态的，this时钟指向函数声明时所在作用域下的this值,call,apply都改变不了this的指向
  function getName() {
    console.log( this.name )
  }
  let getName2 = () => {
    console.log( this.name )
  }

  // 设置window对象的name值
  window.name = '大象'
  const school = {
    name: '想想想做你的王'
  }

  // 直接调用
  // getName()// 大象
  // getName2()// 大象

  // call方法调用
  // getName.call(school)// 想想做你的王
  // getName2.call(school)// 大象


  // 2. 不能作为构造函数实例化对象
  // let Person = (name,age) => {
    // this.name = name
    // this.age = age
  // }
  // let me = new Person('xiao', 20)
  // console.log( me )// Person is not a constructor

  // 3. 不能使用 arguments 变量
  // let fn = () => {
    // console.log( arguments )
  // }
  // fn(1,2,3)// arguments is not defined

  // 4. 箭头函数的再简写
    // 4_1) 省略小括号,当形参有且只有一个的时候
    // let add = num => {
      // return num + num
    // }
    // console.log( add(9) )
    // 4_2) 省略花括号，当代码体只有一条语句的时候,return必须省略，而且语句的执行结果就是函数的返回值
    // let pow = (num) => num*num
    // console.log( pow(10) )

# rest参数的介绍和使用

  // es6引入rest参数，用于获取函数的实参，用来代替arguments
  // es5获取实际参数的方式
  // function date () {
    // console.log( arguments )// 伪数组
  // }
  // date('大象', '白猪', '蓝山')

  // es6 获取实际参数的方式 rest
  function date(...args) {
    console.log( args )//是数组
  }
  date('在石', '子龙', '有毛')
  // 注意：rest参数必须放到参数的最后
  function fn(a,b,...args) {
    console.log( a )// 1
    console.log( b )// 2
    console.log( args )// [3, 4, 5, 6]
  }

  fn(1,2,3,4,5,6)
