# es8async函数介绍

  // async函数
  async function fn () {
    // 返回值是promise对象,返回字符串，数字，undefined，只要你不返回promise，就是成功的返回resolve
    // return 'SSG'

    // 报错，是失败的promisereject
    // throw Error('err;')

    // 返回的结果如果是一个promise对象
    return new Promise((resolve, reject) => {
      // 这里成功了， fn () 就成功了，这里的数据就是fn的返回值数据
      resolve('成功的数据')

      // 失败同理
      // reject('失败的数据')
    })
  }
  const result = fn()
  // console.log( result )// Promise {<fulfilled>: 'SSG'}

  // 调用
  result.then(function ( value ) {
    console.log( value )
  }, function ( reason ) {
    console.log( reason )
  })

# await表达式

1. await 必须放在async函数中
2. await 右侧的表达式一般为promise对象
3. await 返回的是promise成功的值
4. await 的 promise 失败了，就会抛出异常，需要通过try catch去抓捕处理
1
2
3
4
  const p = new Promise((resolve,reject) => {
    // resolve('用户数据')
    resolve('失败了')
  })

  // await 要放在async中
  async function main () {
    try {
      let result =  await p
      console.log( result )
    } catch (error) {
      console.log( error )
    }
  }
  main()

# await读取文件内容

const fs = require('fs')

// 1. 读取【静夜思】
function readJYS() {
  return new Promise((resolve, reject) => {
    fs.readFile('./resources/静夜思.md', (err, data) => {
      // 失败
      if (err) reject(err)
      // 成功
      resolve(data)
    })
  })
}

function readWzry() {
  return new Promise((resolve, reject) => {
    fs.readFile('./resources/Wzry.md', (err, data) => {
      // 失败
      if (err) reject(err)
      // 成功
      resolve(data)
    })
  })
}

function readLoL() {
  return new Promise((resolve, reject) => {
    fs.readFile('./resources/LoL.md', (err, data) => {
      // 失败
      if (err) reject(err)
      // 成功
      resolve(data)
    })
  })
}

  const result2 = await readLoL()
  console.log( result2.toString() )

  const result3 = await readWzry()
  console.log( result3.toString() )
}

# await发送ajax

  // 封装
  // 发送ajax, 返回primise对象
  function sendAjax(url) {

    return new Promise((resolve, reject) => {
      // 1. 创建对象
      const xhr = new XMLHttpRequest()

      // 2. 初始化
      xhr.open('GET', url)

      // 3. 发送请求
      xhr.send()

      // 4. 事件绑定
      xhr.onreadystatechange = function (  ) {
        if ( xhr.readyState === 4) {
          if (xhr.status >= 200 && xhr.status < 300) {
            // 成功
            resolve(xhr.response)
          } else {
            // 失败
            reject(xhr.status)
          }
        }
      }
    })


}

# es8对象方法扩展

  // 声明对象
  const school = {
    name: '想想想做你的王',
    cities: ['beijing', 'sh', 'sz'],
    xuekes: ['front', 'back', 'java']
  }

  // 获取对象所有的键
  console.log( Object.keys(school) )
  /* 
  Array(3)
0: "name"
1: "cities"
2: "xuekes" */

// 获取对象所有的值
console.log( Object.values(school) )//  ['想想想做你的王', Array(3), Array(3)]

// entries 入口
console.log( Object.entries(school) )
/* 
(3) [Array(2), Array(2), Array(2)]
0: (2) ['name', '想想想做你的王']
1: (2) ['cities', Array(3)]
2: (2) ['xuekes', Array(3)] */

// 创建Map
const map = new Map(Object.entries(school))
console.log( map )
/* 
 {"name" => "想想想做你的王"}
1: {"cities" => Array(3)}
2: {"xuekes" => Array(3)} */

console.log( map.get('cities') )//['beijing', 'sh', 'sz']
