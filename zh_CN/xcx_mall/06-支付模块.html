pay/index.js

/*
1 页面加载的时候
  1 从缓存中获取购物车数据，渲染到页面中
    这些数据 checked=true
2 微信支付
  1 那些人 那些账号 可以实现微信支付
    1 企业账号
    2 企业账号的小程序后台中 必须 给 开发者添加上白名单
      1 一个appid，可以绑定多个开发者（必须加入白名单）
      2 这些开发者就可以共用多个开发者
2.5 支付流程：1.创建订单，2.准备预支付，3.发起微信支付，4.查询订单
3 支付按钮
  1 先判断缓存中有没有token
  2 没有就先跳转到授权页面,进行获取token,wx-login
  3 有token，create订单...
  4. 创建订单，获取订单编号
  5. 已经完成微信支付
  6. 手动删除缓存中 已经被选中的商品
  7. 重新把删除后的购物车 填充回缓存
  8. 再跳转页面
*/

import { request } from "../../request/index.js";
import regeneratorRuntime from "../../lib/runtime/runtime.js"
import { getSetting, chooseAddress, openSetting, showModal, showToast, wxPayment } from "../../utils/asyncWx.js";

Page({

  /**
   * 页面的初始数据
   */
  data: {
    //2-1地址对象
    address: {},
    //购物车item数组
    cart: [],
    //全选flag
    allChecked: false,
    //总数量
    totalNum: 0,
    //总价格
    totalPrice: 0
  },
  //2-2每次点这个页面都会调用的函数onShow
  onShow() {
    // console.log('onShow..');

    // 2-2-1 获取wx缓存中收货地址 
    let address = wx.getStorageSync('address');
    // 2-2-4 获取缓存中的购物车数据
    let cart = wx.getStorageSync('cart')||[];
    // 过滤后的购物车数组
    cart = cart.filter(v=>v.checked)

    //3-1总价格总数量
    let totalPrice = 0
    let totalNum = 0
    cart.forEach(item => {
        totalPrice+=item.goods_price * item.nums
        totalNum+=item.nums
    });

    this.setData({
        cart,
        totalPrice,
        totalNum,
        address
      })

  },
  // 点击 支付
  async handleOrderPay() {
    

    try {
      // 1 判断缓存中有没有token
      const token = wx.getStorageSync('token');
      // 2 判断
      if(!token) {
        wx.navigateTo({
          url: '/pages/auth/index'
        });
        return;
      }
      // console.log( '已经存在token' )
      // 3 创建订单
      // 3.1 准备 请求头 参数
      // const header = {Authorization: token}
      // 3.2 准备 请求体 参数
      const order_price = this.data.totalPrice
      const consignee_addr = this.data.address.all
      const cart = this.data.cart
      // console.log( cart )
      // 3.3 定义空数组, 组装订单数组
      let goods = []
      cart.forEach(v => goods.push({
        goods_id: v.goods_id,
        goods_number: v.goods_number,
        goods_price: v.goods_price
      }));
      // console.log( goods )
      const orderParams = { order_price,consignee_addr,goods }
      // 4 准备发送请求,创建订单，获取订单编号
      const res  = await request({
        url: '/my/orders/create',
        method: 'post',
        data: orderParams
      })
      // 在这里创建了订单，拿到了订单编号
      // console.log( res ) // order_id: 1213, user_id: 23, order_number: "HMDD20220223000000001213", order_price: 5999, order_pay: "0", …}
      const { order_number } = res
      // console.log( order_number )

      // 发起 预支付接口
      // const resp = await request({
      const { pay } = await request({
        url: '/my/orders/req_unifiedorder',
        method: 'post',
        data: {order_number}
      })
      // console.log( resp )
      /* 
        pay: nonceStr: "dooBFGq4v4HeN53z"
        package: "prepay_id=wx230936046176145197577f199e8cdf0000"
        paySign: "98D8DB19832C7A028956B8C50C4C580B"
        signType: "MD5"
        timeStamp: "1645580165" */
        
      // 6 发起微信支付,这次是真不行了，注释掉这段api，直接支付成功，删除已购买的商品item
      // ????
      // const response = await wxPayment(pay)
      // ????
      // console.log( response ) // {errMsg: "requestPayment:fail no permission"} // 没权限，还是因为没有企业id，草,

      // 7 查询私服后台订单状态
      /* const response = await request({
        url: '/my/orders/chkOrder',
        method: 'post',
        header,
        data: {order_number}
      })
      console.log( response ) */
      // null, 因为前一步支付没权限,也一样注释掉就好了

      // 随便打印一个支付成功
      console.log( 'zwq:支付成功' )

      await showToast({title: '支付成功'})

      // 8 手动删除缓存中已经支付的商品
      let newCart = wx.getStorageSync('cart');
      // 留下未被选中的商品
      newCart = newCart.filter(v=>!v.checked)
      wx.setStorageSync('cart', newCart);

      // 支付成功后跳转到订单页面
      wx.navigateTo({
        url: '/pages/order/index'
      })
      
    } catch (error) {
      console.log( 'pay err' + error )
      await showToast({title: '支付失败'})
    }



  }
})

/**
 * promise 形式 getSetting
 */
export const getSetting = () => {
  return new Promise((resolve, reject) => {
    wx.getSetting({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}

/**
 * promise 形式 chooseAddress
 */
 export const chooseAddress = () => {
  return new Promise((resolve, reject) => {
    wx.chooseAddress({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}

/**
 * promise 形式 openSetting
 */
 export const openSetting = () => {
  return new Promise((resolve, reject) => {
    wx.openSetting({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}


/**
 * promise 形式 showModal
 */
 export const showModal = ({content}) => {
  return new Promise((resolve, reject) => {
    wx.showModal({
      title: '',
      content: content,
      success: (res) => resolve(res),
      fail: (err)=>reject(err)
    });
  })
}

/**
 * promise 形式 showToast
 */
 export const showToast = ({title}) => {
  return new Promise((resolve, reject) => {
    wx.showToast({
      title: title,
      icon: 'none',
      mask: false,
      success: res=>resolve(res),
      fail: (err)=>reject(err)
    });
  })
}

/**
 * promise 形式 微信登录
 */
 export const wxLogin = () => {
  return new Promise((resolve, reject) => {
    wx.login({
      timeout:10000,
      success: (result) => {
        resolve(result)
      },
      fail: (err) => {
        reject(err)
      }
    });
  })
}

/**
 * Promise形式的小程序的wx支付
 * @param {Object} pay : 支付所必要的参数
 */
 export const wxPayment = (pay) => {
  return new Promise((resolve, reject) => {
    wx.requestPayment({
      ...pay,
      success: (result) => {
        resolve(result)
      },
      fail: (err) => {
        reject(err)
      }
    });
      
  })
}
