4 改写getCates方法接受数据

import { request } from "../../request/index.js";

Page({
	......
  /* 获取分类数据 */
  getCates() {
    request({
      url: "https://api-hmugo-web.itheima.net/api/public/v1/categories"
    }).then((res) => {

      /* console.log(result); */
      this.Cates = res.data.message
      /* 构造左边的菜单数据 */
      let leftMenuList = this.Cates.map(item => item.cat_name)
      /* 构造右边的商品数据 */
      let rightContent = this.Cates[0].children
      /* 把数据返回出去给模板层使用 */
      this.setData({
        leftMenuList,
        rightContent
      })

    })
  }
});

3 完善category/index.js

import { request } from "../../request/index.js";

Page({
  data: {
    /* 左侧菜单的数组 */
    leftMenuList: [],
    /* 右侧商品的数组 */
    rightContent: [],
    /* 当前索引 */
    currentIndex: 0
  },
  /* 定义全局数组：接口的返回数据 */
  Cates: [],
  onLoad: function(options) {

    /**
     *  0 web中的本地存储核小程序中的本地存储的区别
     *    1 写代码的方式不一样了
     *      web: localStorage.setItem('key','value') localStorage.getItem('key')
     *      小程序：wx.setStorageSync('k','v') wx.getStorageSync('k')
     *    2 存的时候，有没有做类型转换
     *      web: 不管存入的数据是什么类型的数据，最终都会ToString
     *      小程序：不存在类型转换的这个操作，存什么类型的数据进去，取出来就是什么类型
     *  1 先判断一下本地存储中有没有旧的数据
     *  2 没有旧的数据，直接发送新的请求
     *  3 有旧的数据，同时旧的数据没有达到过期时间，就使用本地存储的旧数据即可
     */
    /* 1 获取本地存储中的数据 */
    const Cates = wx.getStorageSync("mycates");
    /* 2 判断 */
    if (!Cates) {
      //不存在，发送新的请求
      this.getCates()
    } else {
      //有旧的数据，定义过期时间 10s 改成 5分钟
      if (Date.now()-Cates.time>1000*10) {
        // 重新发送请求
        this.getCates()
      } else {
        console.log('可以使用旧的数据');
        // this.Cates=Cates.data
      }
    }


  },
  /* 获取分类数据 */
  getCates() {
    request({
      url: "https://api-hmugo-web.itheima.net/api/public/v1/categories"
    }).then((res) => {

      /* console.log(result); */
      this.Cates = res.data.message

      /* 把接口中的数据存入到小程序本地存储中 */
      wx.setStorageSync("mycates", {time: Date.now(), data:this.Cates});
