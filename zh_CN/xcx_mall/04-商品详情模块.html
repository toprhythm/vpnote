2 js


/*
1 发送请求获取数据
2 点击轮播图预览大图
  1 给轮播图绑定点击事件
  2 调用小程序的api,previewImage
*/
import { request } from "../../request/index.js";
import regeneratorRuntime from "../../lib/runtime/runtime.js"
Page({


  data: {
    goodsObj: {}
  },

  //3-2全局商品对象，为了拿到urls
  GoodsInfo:{},

  onLoad: function (options) {
    const { goods_id } = options
    this.getGoodsDetail(goods_id)
  },

  //获取商品详情
  async getGoodsDetail(goods_id) {
    const goodsObj = await request({url: "/goods/detail", data: {goods_id}})
    //3-3复制给全局商品对象
    this.GoodsInfo = goodsObj
    this.setData({
      goodsObj: {
        goods_name:goodsObj.goods_name,
        goods_price:goodsObj.goods_price,
        goods_introduce:goodsObj.goods_introduce.replace(/\.webp/g, '.jpg'),
        pics:goodsObj.pics
      }
    })
  },

  // 点击轮播图预览
  handlePreviewImage (e) {
    // console.log('pre');

    const current = e.currentTarget.dataset.url

    // const { index } = e.dataset

    // 3-1 构造要预览的图片数组
    const urls = this.GoodsInfo.pics.map(item => item.pics_mid)

    wx.previewImage({
      current: current,
      urls: urls
    });
      

  }


})

完整的handleCardAdd

  handleCardAdd (e) {
    // console.log('gouwuche');

    // 1 获取缓存中的购物车数组
    let cart = wx.getStorageSync("cart") || []
    // 2 判断商品对象是否存在于购物车数组
    let index = cart.findIndex(item => item.goods_id === this.GoodsInfo.goods_id)
    if (index===-1) {
      //3不存在，第一次添加
      this.GoodsInfo.nums=1
      cart.push(this.GoodsInfo)
    } else {
      //4已经存在，nums++
      cart[index].nums++
    }
    //5 把购物车重新返回数组
    wx.setStorageSync('cart', cart);
    // 6 弹窗提示添加成功
    wx.showToast({
      title: '加入成功',
      icon: 'success',
      //防止用户手抖疯狂点击按钮,点一次必须1.5 s之后才能再次点击成功
      mask: true
    });
      
      
      

  }
