# 03-完整的获取收货地址

有一个机制，你第一次询问用户拿到地址的时候用户如果点击取消，那么authSetting.scope.address=false，下一次用户后悔，想点击获取地址，会发现无法点击

1 cart/index.js

/* 
1 获取用户的收货地址
  1 绑定点击事件
  2 调用内置api获取用户收货地址 wx.chooseAdress
2 获取用户对小程序所授予获取地址的权限状态 scope
  1 假设用户点击获取提示框中的确定
    scope=true
  2 假设用户点击取消
    scope=false
  3 假设用户从来没有调用过收货地址的api
    scope=undefined
    1 引诱用户自己打开授权设置页面，当用户重新给与获取地址权限的时候
    2 调用获取地址api
*/
Page({

  /**
   * 页面的初始数据
   */
  data: {

  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {

  },
  //获取真实地址
  handleChooseAddress () {

    // 1 获取权限状态
    wx.getSetting({
      success: (result) => {
        // 2 获取权限状态 主要发现一些属性名很怪异的时候 都要使用[]形式来获取属性值
        const scopeAdress = result.authSetting["scope.address"]
        if (scopeAdress===true||scopeAdress===undefined) {
          wx.chooseAddress({
            success: (result1) => {
              console.log(result1);
            }
          });
        } else {
          // 3 用户以前拒绝过授予权限，先引诱用户打开授权页面
          wx.openSetting({
            success: (result2) => {
              //  可以调用收货地址代码
              wx.chooseAddress({
                success: (result3) => {
                  console.log(result3);
                }
              });
            }
          });
        }
      }
    });
    
      
  }
})

/**
 * promise 形式 getSetting
 */
export const getSetting = () => {
  return new Promise((resolve, reject) => {
    wx.getSetting({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}

/**
 * promise 形式 chooseAddress
 */
 export const chooseAddress = () => {
  return new Promise((resolve, reject) => {
    wx.chooseAddress({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}

/**
 * promise 形式 openSetting
 */
 export const openSetting = () => {
  return new Promise((resolve, reject) => {
    wx.openSetting({
      success: result => resolve(result),
      fail: err => reject(err)
    });
  })
}

/* 
1 获取用户的收货地址
  1 绑定点击事件
  2 调用内置api获取用户收货地址 wx.chooseAdress
2 获取用户对小程序所授予获取地址的权限状态 scope
  1 假设用户点击获取提示框中的确定
    scope=true
  2 假设用户点击取消
    scope=false
  3 假设用户从来没有调用过收货地址的api
    scope=undefined
    1 引诱用户自己打开授权设置页面，当用户重新给与获取地址权限的时候
    2 调用获取地址api
*/

import { request } from "../../request/index.js";
import regeneratorRuntime from "../../lib/runtime/runtime.js"
import { getSetting, chooseAddress, openSetting } from "../../utils/asyncWx.js";

Page({

  /**
   * 页面的初始数据
   */
  data: {

  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {

  },
  //获取真实地址
  async handleChooseAddress () {

    try {
      // aysnc优化过的 
      // 1 获取权限状态
      const res1 = await getSetting()
      const scopeAddress = res1.authSetting['scope.address']
      // 2 判断权限状态是否为true
      if (scopeAddress===false){
        await openSetting()
      }
      // 4 调用收货地址api
      const address = await chooseAddress()
      // 5 存入wx缓存中
      wx.setStorageSync('address', address);
    } catch (error) {
      console.log(error);
    }
      
      
  }
})

  2 获取缓存中数组cart并且setData
4 全选的实现
  1 onshow 获取缓存中的购物车数组
  2 根据购物车中的商品数据，所有的商品都被选中 checked=true 全选就被选中
5 总价格和总数量
  1 都需要商品被选中 我们才拿它来计算
  2 获取购物车数组
  3 遍历
  4 判断商品是否被选中
  5 总价格 += 商品的单价 * 商品的数量
  6 把计算后的价格和数量设置回data
6 商品的选中
  1 绑定change事件
  2 获取到被修改的商品对象
  3 商品对象的选中状态取反
  4 重新填重回data中和缓存中
  5 重新计算全选，总价格，总数量
*/

Page({

  /**
   * 页面的初始数据
   */
  data: {
    //2-1地址对象
    address: {},
    //购物车item数组
    cart: [],
    //全选flag
    allChecked: false,
    //总数量
    totalNum: 0,
    //总价格
    totalPrice: 0
  },
  //2-2每次点这个页面都会调用的函数onShow
  onShow() {
    // console.log('onShow..');

    // 2-2-1 获取wx缓存中收货地址 
    let address = wx.getStorageSync('address');
    // 2-2-4 获取缓存中的购物车数据
    const cart = wx.getStorageSync('cart')||[];
    
    this.setCart(cart)
    this.setData({address})
      

  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {

  },
  //获取真实地址
  async handleChooseAddress () {

    try {
      // aysnc优化过的 
      // 1 获取权限状态
      const res1 = await getSetting()
      const scopeAddress = res1.authSetting['scope.address']
      // 2 判断权限状态是否为true
      if (scopeAddress===false){
        await openSetting()
      }
      // 4 调用收货地址api
      const address = await chooseAddress()
      // 5 存入wx缓存中
      wx.setStorageSync('address', address);
    } catch (error) {
      console.log(error);
    }
      
      
  },
  //商品的点击选择框
  handleItemChange (e) {
    
    const goods_id = e.currentTarget.dataset.id
    // console.log(goods_id);

    let {cart} = this.data
    let index = cart.findIndex(item=>item.goods_id===goods_id)
    cart[index].checked=!cart[index].checked
    
    this.setCart(cart)

  },
  //setCart
  setCart (cart) {
    

    let allChecked = true
    //3-1总价格总数量
    let totalPrice = 0
    let totalNum = 0
    cart.forEach(item => {
      if (item.checked) {
        totalPrice+=item.goods_price * item.nums
        totalNum+=item.nums
      } else {
        //任意一个item.checked为false,allChecked就是false
        allChecked = false
      }
    });
    //判断cart数组是否为空
    allChecked = cart.length!=0?allChecked:false

    this.setData({cart,totalPrice,totalNum,allChecked})

    wx.setStorageSync('cart', cart)
  }
})
