/**
 * 上拉加载下一页
 * 用户上滑页面，滚动条触底，开始加载下一页数据
 * 1 找到滚动条触底事件
 * 2 判断还有没有下一页数据
 *  1 获取到总页数 Math.ceil(总条数 / 页容量pagesize) //向上取整 // Math(23 /10 ) 3
 *  2 获取到当前页码 pagenum
 *  3 判断一下 当前页面是否大于等于总页码，表示没有下一页数据
 * 3 假设没有下一页数据，弹出一个提示
 * 4 假设还有下一页数据，就加载下一页的数据
 *  1 当前页码++
 *  2 重新发送请求
 *  3 数据请求回来 要对data中的goodsList进行拼接而不是全部替换
 */
import { request } from "../../request/index.js";
import regeneratorRuntime from "../../lib/runtime/runtime.js"
Page({

  /**
   * 页面的初始数据
   */
  data: {
    // 商品列表
    goodsList: [],
  	......
  },
  //全局查询参数对象
  QueryParams: {
    query: "",
    cid: "",
    pagenum: 1,
    pagesize: 10
  },
  //全局的总页数
  totalPages: 1,

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {
    //console.log(options);
    this.QueryParams.cid=options.cid;
    this.getGoodsList()
  },

  async getGoodsList() {
    const res = await request({url:'/goods/search', data:this.QueryParams})

    const total = res.total;
    this.totalPages = Math.ceil(total / this.QueryParams.pagesize)
    // console.log(this.totalPages);

    this.setData({
      goodsList:[...this.data.goodsList, ...res.goods]
    })
  },

......

  onReachBottom(){
    // console.log('页面触底');

    //1判断还有没有下一页数据
    if(this.QueryParams.pagenum>=this.totalPages) {
      //没有下一页数据

  	  wx.showToast({

      });
  
    } else {
      // console.log('还有下一页数据');

      this.QueryParams.pagenum++;
      this.getGoodsList()

    }

  }

})

3 删除掉onload的里的showloading，在request函数里用loading

/**
 * 为什么要有requestCount
 * 因为，首页的onload，一次有三个请求，swiper，导航，floor，三次axios意味着要loading三次，太傻比了，
 * 我们要让它只loading一次，三个请求全结束，那一次loading结束
 * 1 定义一个请求数变量默认为0
 * 2 每发送一个请求请求数+1
 * 3 没接收一次响应请求数-1
 * 4 当请求数等于0的时候，可以关闭loading效果
 */
let requestCount = 0;

export const request = (params) => {
  
  requestCount++;

  /* loading 开始 */
  wx.showLoading({
    title: '加载中',
    mask: true,
  });

  /* 1 定义公共url */
  const baseUrl = "https://api-hmugo-web.itheima.net/api/public/v1"
  return new Promise((resolve, reject) => {
    wx.request({
      ...params,
      /* 2 拼接url */
      url: baseUrl+params.url,
      success: (result) => {
        resolve(result.data.message)
      },
      fail: err => reject(err),
      complete: function() {
        requestCount--;
        if(requestCount===0) {
          wx.hideLoading();
        }
      }
    });
  })
}
