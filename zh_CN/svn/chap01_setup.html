# 02-为什么要使用svn


	从个人开发到团队合作

	服务器 a1.py内容: aabbccdd;

	小红 								小蓝
	修改aa22ccdd				修改aabbcc44
	
	小红先上传，v1
	小蓝后上传, v2
	
	普通的文件服务器：后上传的文件覆盖先上传的文件
	
	svn服务器：会做冲突，请你解决文件冲突
	
	所以最后文件是aa22cc44, v3
	
	# svn和普通服务器的区别: 深入文件内容以行为单位更精细的区管理文件内容的变化

# 03-版本控制工具的功能


- 协同修改
	- 多人并行不悖的修改服务器端的同一个文件
- 数据备份
	- 如果本地文件发生丢失可以服务器端文件进行恢复
- 增量式的版本管理
	- 服务器端保存的每一个版本信息只保存有修改的局部内容,节约服务器端资源
- 权限控制
	- 对团队中参与开发的人员进行权限控制
- 历史记录
	- 查看修改人，修改时间，修改的内容，日志信息
	- 将本地文件恢复到某个历史状态(时光机)

			三个操作：checkout（检出），update（更新），commit提交
			其中检出和更新是服务器对客户端的操作，提交是客户端对服务器的操作

 # 提交commit：把本地的修改上传到服务器

# 10-让svn开机自启动

chkconfig|grep svn # 查看svnserve的自启状态,还是和视频教程不一致

chkconfig svnserve on # 开机自启svn

# 老师应该是怎么centos6上进行的操作

cd /etc/rc.d/init.d/
ll 
...
svnserve # 对应可执行脚本文件的路径
...

cp svnserve svnserve.bak

vim /etc/rc.d/init.d/svnserve
esc:set nu 32
旧的arg: args="--daemon --pid-file=${pidfile} $OPTIONS"
新的arg: args="--daemon --root /var/svn/repository --listen-port 2255 --pid-file=${pidfile} $OPTIONS"

# 看下2255端口上有没有占用服务
netstat -anp|grep 2255

# 启动svn服务
service svnserve start

# 查看svn状态
service svnserve status

# 11-b站大神搞的svn一分钟搭建到连接



yum install subversion -y

mkdir -pv /var/svn/repository

svnadmin create /var/svn/repository

cd /var/svn/repository/conf

vi authz
[/]
toprhythm=rw

vim passwd
toprhythm=toprhythm

vim svnserve.conf
auth-access = write #解开注释 
password-db = passwd #解开注释
authz-db = authz #解开注释
realm = /var/svn/repository #解开注释

svnserve -d -r /var/svn/repository

# 14-harry做提交操作


cd harry/

vim hello.txt 1232klsadjklsajhello

[root@localhost harry]# svn commit hello.txt
svn: E200009: 提交失败(细节如下): 
svn: E200009: “/root/workspace/harry/hello.txt” 尚未纳入版本控制

# svn要求提交一个新建的文件之前先把，必须先纳入版本控制体系

svn add hello.txt

[root@localhost harry]# svn commit hello.txt
svn: E205007: 提交失败(细节如下): 
svn: E205007: 无法使用外部编辑器获得日志信息；考虑设置环境变量 $SVN_EDITOR，或者使用 --message (-m) 或 --file (-F) 选项
svn: E205007: 没有设置 SVN_EDITOR，VISUAL 或 EDITOR 环境变量，运行时的配置参数中也没有 “editor-cmd” 选项

# 要求一：附加日志信息，写java代码时的注释，用来标记本次操作所做的修改
svn commit -m 'xxx' hello.txt

cd conf/
cp svnserve.conf svnserve.conf.bak
vim svnserve.conf
把 anno-access = write # 匿名访问打开，赋予写权限，我们不开，我们用户认证都做好了
